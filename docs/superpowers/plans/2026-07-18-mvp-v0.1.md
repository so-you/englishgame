# MVP v0.1 实现计划：英语爬塔（核心验证版）

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 spec v3 的 MVP v0.1：固定三战流程（2 普通战 + 1 Boss）的英语学习卡牌战斗，含战前教学、过载、专注、语流、勇气奖励、完整 SRS 状态机、CSV/TSV 导入向导、IndexedDB 存档与无障碍设置。

**Architecture:** 纯 TS 领域核心（`src/core/`，无 React/DOM/IndexedDB 依赖，时间与随机数注入）+ 薄 React UI 层。双层卡牌模型：CardTemplate（战斗躯壳）与 LearningUnit（学习铭文）在会话开始时绑定为 CombatCardInstance。SRS 计分资格由核心模块计算，UI 只提交作答事实。

**Tech Stack:** Vite + React + TypeScript（strict）、Zustand、framer-motion、Vitest、idb、papaparse、fake-indexeddb（测试）、Web Speech API。

**Spec:** `docs/superpowers/specs/2026-07-18-vocab-roguelike-design.md`（v3 最终版）

## Global Constraints

来自 spec 第 15 节「最终决策清单」，实现中不得自行改回：

- 双层卡牌模型（CardTemplate + LearningUnit → CombatCardInstance）
- 卡牌基础效果不读取掌握度；答错不哑火、不退能量、无生命惩罚
- 新内容先在安全界面（战前）教学；战斗中不弹首次教学面板
- 只有到期窗口内第一次符合条件的作答可改变 SRS（`lastGradedDueAt !== dueAt`）

- 一个词义 = 一个原子 VocabularyUnit；类型必须能表达语法练习（MVP 不做语法内容）
- 过载消耗每场有限专注（3 点，不每回合刷新）

- 敌人与遗物不能修改练习答案、计分资格或无障碍设置
- 所有实验参数集中在 `src/core/battle/balance.ts`

- 持久化数据带 `schemaVersion`；SRS 状态更新与计分日志在同一事务写入

```ts
export function normalizeDisplay(input: string): string {
  return input.normalize('NFC').trim().replace(/\s+/g, ' ')
}

export function normalizeForMatch(input: string): string {
  return normalizeDisplay(input).toLowerCase()
}

export function levenshtein(a: string, b: string): number {
  const m = a.length, n = b.length
  const dp: number[] = Array.from({ length: n + 1 }, (_, j) => j)
  for (let i = 1; i <= m; i++) {
    let prev = dp[0]
    dp[0] = i
    for (let j = 1; j <= n; j++) {
      const tmp = dp[j]
      dp[j] = Math.min(dp[j] + 1, dp[j - 1] + 1, prev + (a[i - 1] === b[j - 1] ? 0 : 1))
      prev = tmp
    }
  }
  return dp[n]
}
```

`src/core/learning/answer.ts`:

```ts
import type { Exercise } from './types'
import { normalizeDisplay, normalizeForMatch, levenshtein } from './text'

export interface AnswerResult {
  correct: boolean
  suggestConfirm?: boolean
}

export function checkAnswer(exercise: Exercise, response: string | string[]): AnswerResult {
  const spec = exercise.answer
  if (spec.kind === 'choice') {
    return { correct: response === spec.correctOptionId }
  }
  if (spec.kind === 'text') {
    const raw = Array.isArray(response) ? response[0] ?? '' : response
    const norm = spec.caseSensitive ? normalizeDisplay(raw) : normalizeForMatch(raw)
    const accepted = spec.accepted.map(a => (spec.caseSensitive ? normalizeDisplay(a) : normalizeForMatch(a)))
    if (accepted.includes(norm)) return { correct: true }
    const targets = spec.accepted.map(a => normalizeForMatch(a))
    const minDist = Math.min(...targets.map(t => levenshtein(normalizeForMatch(raw), t)))
    if (norm.length > 0 && minDist === 1) return { correct: false, suggestConfirm: true }
    return { correct: false }
  }
  const resp = (Array.isArray(response) ? response : [response]).map(normalizeForMatch)
  const ok = spec.acceptedOrders.some(
    order => order.length === resp.length && order.every((t, i) => normalizeForMatch(t) === resp[i]),
  )
  return { correct: ok }
}
```

```ts
import type { ContentPack, Exercise, ExerciseKind, Mastery } from './types'

export function exerciseKindsForMastery(mastery: Mastery): ExerciseKind[] {
  switch (mastery) {
    case 0: return []
    case 1: return ['vocab-meaning-choice']
    case 2: return ['vocab-word-choice']
    case 3: return ['vocab-spelling']
    case 4: return ['vocab-listening-spelling', 'vocab-spelling']
    case 5: return ['vocab-spelling', 'vocab-word-choice', 'vocab-meaning-choice']
  }
}

export function pickExerciseForUnit(
  pack: ContentPack,
  unitId: string,
  mastery: Mastery,
  recentExerciseIds: string[],
  opts?: { kinds?: ExerciseKind[] },
): Exercise | undefined {
  const kinds = opts?.kinds ?? exerciseKindsForMastery(mastery)
  const candidates = pack.exercises.filter(e => e.unitId === unitId && kinds.includes(e.kind))
  if (candidates.length === 0) return undefined
  const fresh = candidates.filter(e => !recentExerciseIds.includes(e.id))
  const pool = fresh.length > 0 ? fresh : candidates
  return pool[0]
}
```

```ts
export function isEligibleForGrading(state: SrsState, exercise: Exercise, now: number): boolean {
  if (state.mastery < 1) return false
```

```ts
import { DAY_MS, INTERVAL_DAYS, type Mastery, type RecallGrade, type SrsState } from './state'

export function gradeOf(correct: boolean, usedHint: boolean): RecallGrade {
  if (!correct) return 'again'
  return usedHint ? 'hard' : 'good'
}

export function applyGrade(state: SrsState, grade: RecallGrade, now: number): SrsState {
  const base: SrsState = { ...state, lastReviewedAt: now, lastGradedDueAt: state.dueAt }
  if (grade === 'good') {
    const mastery = Math.min(5, state.mastery + 1) as Mastery
    const intervalDays = INTERVAL_DAYS[mastery]
    return { ...base, mastery, consecutiveCorrect: state.consecutiveCorrect + 1, intervalDays, dueAt: now + intervalDays * DAY_MS }
  }
  if (grade === 'hard') {
    return { ...base, dueAt: now + state.intervalDays * DAY_MS }
  }
  const mastery = Math.max(1, state.mastery - 1) as Mastery
  return { ...base, mastery, consecutiveCorrect: 0, lapses: state.lapses + 1, intervalDays: 1, dueAt: now + DAY_MS }
}
```

```ts
  exercise: Exercise
  correct: boolean
  usedHint: boolean
  responseMs?: number
  sessionId: string
  encounterId?: string
  now: number
}

/** 战斗 UI 只能提交作答事实；graded 资格由此处计算（spec 6.2） */
export function recordRecall(input: RecordRecallInput): { state: SrsState; log: Omit<RecallLog, 'id'> } {
  const eligible = isEligibleForGrading(input.state, input.exercise, input.now)
  const grade = eligible ? gradeOf(input.correct, input.usedHint) : undefined
  const state = eligible && grade ? applyGrade(input.state, grade, input.now) : input.state
  const log: Omit<RecallLog, 'id'> = {

    exerciseId: input.exercise.id,
    sessionId: input.sessionId,
    encounterId: input.encounterId,
    scheduledDueAt: input.state.dueAt,
    kind: input.exercise.kind,
    correct: input.correct,
    usedHint: input.usedHint,
    responseMs: input.responseMs,
    eligible,
    graded: eligible,
    grade,
    createdAt: input.now,
  }
  return { state, log }
}
```

```ts
import type { VocabularyUnit } from '../learning/types'
import { initialSrsState, type SrsState } from './state'

export interface LearningSetSelection {
  due: string[]
  fresh: string[]
  practice: string[]
  unitIds: string[]
}

export function buildLearningSet(
  units: VocabularyUnit[],
  states: ReadonlyMap<string, SrsState>,
  now: number,
  opts: { newQuota: number; targetSize: number; dueMax: number },
): LearningSetSelection {
  const eligible = units
    .filter(u => u.enrichmentStatus === 'complete' && u.definitionZh.length > 0)
    .map(u => ({ u, s: states.get(u.id) ?? initialSrsState(u.id) }))

  const due = eligible
    .filter(x => x.s.mastery >= 1 && x.s.dueAt !== undefined && x.s.dueAt <= now)
    .sort((a, b) => (a.s.dueAt! - b.s.dueAt!) || (a.s.mastery - b.s.mastery) || (b.s.lapses - a.s.lapses))
    .slice(0, opts.dueMax)

  const dueIds = new Set(due.map(x => x.u.id))
  const fresh = eligible
    .filter(x => x.s.mastery === 0 && !dueIds.has(x.u.id))
    .sort((a, b) => (a.u.frequencyRank ?? Infinity) - (b.u.frequencyRank ?? Infinity) || a.u.id.localeCompare(b.u.id))
    .slice(0, opts.newQuota)

  const chosen = new Set([...dueIds, ...fresh.map(x => x.u.id)])
  const practice = eligible
    .filter(x => !chosen.has(x.u.id) && x.s.mastery >= 1)
    .sort((a, b) => (b.s.lastReviewedAt ?? 0) - (a.s.lastReviewedAt ?? 0))
    .slice(0, Math.max(0, opts.targetSize - due.length - fresh.length))

  const unitIds = [...due, ...fresh, ...practice].map(x => x.u.id)
  return {
    due: due.map(x => x.u.id),
    fresh: fresh.map(x => x.u.id),
    practice: practice.map(x => x.u.id),
    unitIds,
  }
}
```

```ts
export type RngState = number

export function rngNext(s: RngState): [number, RngState] {
  let a = (s + 0x6d2b79f5) | 0
  let t = Math.imul(a ^ (a >>> 15), 1 | a)
  t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t
  const value = ((t ^ (t >>> 14)) >>> 0) / 4294967296
  return [value, a]
}

export function rngInt(s: RngState, min: number, maxExclusive: number): [number, RngState] {
  const [v, ns] = rngNext(s)
  return [min + Math.floor(v * (maxExclusive - min)), ns]
}

export function rngShuffle<T>(s: RngState, items: T[]): [T[], RngState] {
  const arr = [...items]
  let state = s
  for (let i = arr.length - 1; i > 0; i--) {
    const [j, ns] = rngInt(state, 0, i + 1)
    state = ns
    ;[arr[i], arr[j]] = [arr[j], arr[i]]
  }
  return [arr, state]
}
```

```ts
import { BALANCE } from './balance'
import { rngShuffle } from './rng'
import type { BattleState, CombatEffect, Intent, OverchargeOutcome, PlayContext, CardTemplate, CombatCardInstance, EnemyDef } from './types'

export interface CreateBattleInput {
  seed: number
  playerHp: number
  playerMaxHp: number
  deck: CombatCardInstance[]
  enemyDef: EnemyDef
  relics: string[]
  templates: Record<string, CardTemplate>
}

export function createBattle(input: CreateBattleInput): BattleState {
  const cards = Object.fromEntries(input.deck.map(c => [c.id, c]))
  const [drawPile, rngState] = rngShuffle(input.seed, input.deck.map(c => c.id))
  let s: BattleState = {
    seed: input.seed,
    rngState,
    turn: 1,
    player: { hp: input.playerHp, maxHp: input.playerMaxHp, block: 0, statuses: { weak: 0 } },
    enemy: { hp: input.enemyDef.hp, maxHp: input.enemyDef.hp, block: 0, statuses: { weak: 0 }, def: input.enemyDef, intentIndex: 0 },
    energy: BALANCE.energyPerTurn,
    energyBonusNextTurn: 0,
    focus: BALANCE.focusPerBattle,
    fluency: input.relics.includes('fluency-charm') ? 1 : 0,
    cards,
    templates: input.templates,
    drawPile,
    hand: [],
    discardPile: [],
    relics: [...input.relics],
    courageClaimed: [],
    rootCompassUsed: false,
    errorNotebookUsed: false,
    result: null,
    log: [{ type: 'turn', turn: 1 }],
  }
  s = drawCards(s, BALANCE.handSize)
  return s
}

export function getIntent(state: BattleState): Intent {
  return state.enemy.def.intents[state.enemy.intentIndex % state.enemy.def.intents.length]
}

function drawCards(s: BattleState, n: number): BattleState {
  let { drawPile, discardPile, rngState } = s
  const hand = [...s.hand]
  for (let i = 0; i < n; i++) {
    if (hand.length >= BALANCE.maxHand) break
    if (drawPile.length === 0) {
      if (discardPile.length === 0) break
      const [shuffled, rs] = rngShuffle(rngState, discardPile)
      rngState = rs
      drawPile = shuffled
      discardPile = []
    }
    hand.push(drawPile[0])
    drawPile = drawPile.slice(1)
  }
  return { ...s, hand, drawPile, discardPile, rngState }
}

function damageEnemy(s: BattleState, amount: number): BattleState {
  const toBlock = Math.min(s.enemy.block, amount)
  const toHp = amount - toBlock
  return {
    ...s,
    enemy: { ...s.enemy, block: s.enemy.block - toBlock, hp: s.enemy.hp - toHp },
    log: [...s.log, { type: 'damage', source: 'player', amount: toHp }],
  }
}

function damagePlayer(s: BattleState, amount: number): BattleState {
  const toBlock = Math.min(s.player.block, amount)
  const toHp = amount - toBlock
  return {
    ...s,
    player: { ...s.player, block: s.player.block - toBlock, hp: s.player.hp - toHp },
    log: [...s.log, { type: 'damage', source: 'enemy', amount: toHp }],
  }
}

function applyEffect(s: BattleState, fx: CombatEffect, mult: number, weakMult: number): BattleState {
  switch (fx.type) {
    case 'damage':
      return damageEnemy(s, Math.floor(fx.amount * mult * weakMult))
    case 'block':
      return {
        ...s,
        player: { ...s.player, block: s.player.block + Math.floor(fx.amount * mult) },
        log: [...s.log, { type: 'block', source: 'player', amount: Math.floor(fx.amount * mult) }],
      }
    case 'draw':
      return drawCards(s, fx.amount)
    case 'apply-status':
      return {
        ...s,
        enemy: { ...s.enemy, statuses: { weak: s.enemy.statuses.weak + fx.stacks } },
        log: [...s.log, { type: 'status', target: 'enemy', status: 'weak', stacks: fx.stacks }],
      }
  }
}

function checkResult(s: BattleState): BattleState {
  if (s.result) return s
  if (s.enemy.hp <= 0) return { ...s, result: 'victory', log: [...s.log, { type: 'result', result: 'victory' }] }
  if (s.player.hp <= 0) return { ...s, result: 'defeat', log: [...s.log, { type: 'result', result: 'defeat' }] }
  return s
}

export function playCard(
  state: BattleState,
  cardId: string,
  outcome: OverchargeOutcome | null,
  ctx: PlayContext,
): BattleState {
  if (state.result) return state
  const card = state.cards[cardId]
  if (!card || !state.hand.includes(cardId)) return state
  const tpl = state.templates[card.templateId]
  if (state.energy < tpl.cost) return state
  if (outcome !== null && (state.focus < 1 || !card.learningUnitId)) return state

  let s: BattleState = {
    ...state,
    energy: state.energy - tpl.cost,
    focus: outcome !== null ? state.focus - 1 : state.focus,
    hand: state.hand.filter(id => id !== cardId),
    discardPile: [...state.discardPile, cardId],
  }

  // 过载加成倍率（掌握度决定题型与奖励，不决定基础战力）
  let mult = 1
  if (outcome === 'correct' || outcome === 'correct-hint') {
    const mastery = ctx.unitMastery ?? 1
    let bonus = BALANCE.overchargeBonusByMastery[mastery] ?? 0.5
    if (outcome === 'correct-hint') bonus *= BALANCE.hintBonusMult
    if (outcome === 'correct' && ctx.unitDue && mastery <= BALANCE.lowMasteryMax) {
      bonus *= BALANCE.challengeBonusMult
    }
    mult = 1 + bonus
  }

  // 勇气奖励：到期或低掌握度项目，每场第一次尝试即得（无论对错）
  const unitId = card.learningUnitId
  if (outcome !== null && unitId && !s.courageClaimed.includes(unitId)) {
    const lowOrDue = (ctx.unitDue ?? false) || (ctx.unitMastery ?? 5) <= BALANCE.lowMasteryMax
    if (lowOrDue) {
      let courage = BALANCE.courageBlock
      let notebookUsed = s.errorNotebookUsed
      if (s.relics.includes('error-notebook') && !notebookUsed) {
        courage += BALANCE.errorNotebookBonus
        notebookUsed = true
      }
      s = {
        ...s,
        courageClaimed: [...s.courageClaimed, unitId],
        errorNotebookUsed: notebookUsed,
        player: { ...s.player, block: s.player.block + courage },
        log: [...s.log, { type: 'courage', amount: courage }],
      }
    }
  }

  const weakMult = s.player.statuses.weak > 0 ? BALANCE.weakDamageMult : 1
  for (const fx of tpl.effects) {
    s = applyEffect(s, fx, mult, weakMult)
  }

  // 语流与遗物
  if (outcome === 'correct' || outcome === 'correct-hint') {
    let fluency = s.fluency + 1
    let energyBonusNextTurn = s.energyBonusNextTurn
    if (fluency >= BALANCE.fluencyThreshold) {
      fluency = 0
      energyBonusNextTurn += 1
      s = { ...s, log: [...s.log, { type: 'energy-bonus' }] }
    }
    s = { ...s, fluency, energyBonusNextTurn }
    if (outcome === 'correct' && s.relics.includes('root-compass') && !s.rootCompassUsed) {
      s = drawCards({ ...s, rootCompassUsed: true }, 1)
    }
  } else if (outcome === 'wrong') {
    s = { ...s, fluency: Math.max(0, s.fluency - 1) }
  }

  s = { ...s, log: [...s.log, { type: 'card-played', cardId, overcharge: outcome }] }
  return checkResult(s)
}

export function endTurn(state: BattleState): BattleState {
  if (state.result) return state
  // 玩家回合结束：弃手牌，玩家虚弱 -1
  let s: BattleState = {
    ...state,
    discardPile: [...state.discardPile, ...state.hand],
    hand: [],
    player: { ...state.player, statuses: { weak: Math.max(0, state.player.statuses.weak - 1) } },
  }

  // 敌人回合：敌人旧格挡清零，执行当前意图
  const intent = getIntent(s)
  s = { ...s, enemy: { ...s.enemy, block: 0 } }
  const enemyWeakMult = s.enemy.statuses.weak > 0 ? BALANCE.weakDamageMult : 1
  if (intent.type === 'attack') {
    const hits = intent.hits ?? 1
    for (let i = 0; i < hits; i++) {
      s = damagePlayer(s, Math.floor(intent.amount * enemyWeakMult))
    }
  } else if (intent.type === 'attack-block') {
    s = damagePlayer(s, Math.floor(intent.amount * enemyWeakMult))
    s = { ...s, enemy: { ...s.enemy, block: s.enemy.block + intent.block } }
  } else {
    s = {
      ...s,
      enemy: { ...s.enemy, block: s.enemy.block + intent.block },
      player: { ...s.player, statuses: { weak: s.player.statuses.weak + intent.stacks } },
      log: [...s.log, { type: 'status', target: 'player', status: 'weak', stacks: intent.stacks }],
    }
  }
  s = { ...s, enemy: { ...s.enemy, statuses: { weak: Math.max(0, s.enemy.statuses.weak - 1) }, intentIndex: s.enemy.intentIndex + 1 } }
  s = checkResult(s)
  if (s.result) return s

  // 新玩家回合：格挡清零、能量刷新（含语流奖励）、摸牌
  const turn = s.turn + 1
  s = {
    ...s,
    turn,
    player: { ...s.player, block: 0 },
    energy: BALANCE.energyPerTurn + s.energyBonusNextTurn,
    energyBonusNextTurn: 0,
    log: [...s.log, { type: 'turn', turn }],
  }
  s = drawCards(s, BALANCE.handSize)
  return s
}
```

```ts
export function computeSettlement(

  srsStates: ReadonlyMap<string, SrsState>,
  terms: ReadonlyMap<string, { term: string; definitionZh: string }>,

  const attemptedDue = new Set(session.attempts.filter(a => dueSet.has(a.unitId)).map(a => a.unitId))
  const wrongUnits = session.attempts

    .map(a => ({ unitId: a.unitId, term: terms.get(a.unitId)?.term ?? a.unitId, correctAnswer: terms.get(a.unitId)?.definitionZh ?? '' }))

    totalAttempts: session.attempts.length,
    correctAttempts: session.attempts.filter(a => a.correct).length,
    dueTotal: dueSet.size,
    dueAttempted: attemptedDue.size,
    wrongUnits,
    nextReviews: session.learningSet.unitIds.map(id => ({ unitId: id, dueAt: srsStates.get(id)?.dueAt })),
```

```ts
import { playCard, type BattleState } from '../battle/engine'
import type { OverchargeOutcome } from '../battle/types'
import { exerciseKindsForMastery, pickExerciseForUnit } from '../learning/exercise-for'
import type { ContentPack, Exercise } from '../learning/types'
import { recordRecall } from '../srs/record'
import type { Mastery, RecallLog, SrsState } from '../srs/state'

export type PrepareResult =
  | { unitId: string; exercise: Exercise; mastery: Mastery; due: boolean }
  | { error: 'no-unit' | 'no-focus' | 'no-exercise' }

export function prepareOvercharge(input: {
  battle: BattleState
  cardId: string
  pack: ContentPack
  srsStates: ReadonlyMap<string, SrsState>
  speechAvailable: boolean
  recentExerciseIds: string[]
  now: number
}): PrepareResult {
  const card = input.battle.cards[input.cardId]
  if (!card?.learningUnitId) return { error: 'no-unit' }
  if (input.battle.focus < 1) return { error: 'no-focus' }
  const unitId = card.learningUnitId
  const srs = input.srsStates.get(unitId)
  if (!srs) return { error: 'no-unit' }
  let kinds = exerciseKindsForMastery(srs.mastery)
  if (!input.speechAvailable) kinds = kinds.filter(k => k !== 'vocab-listening-spelling')
  if (kinds.length === 0) kinds = ['vocab-spelling']
  const exercise = pickExerciseForUnit(input.pack, unitId, srs.mastery, input.recentExerciseIds, { kinds })
  if (!exercise) return { error: 'no-exercise' }
  const due = srs.dueAt !== undefined && srs.dueAt <= input.now
  return { unitId, exercise, mastery: srs.mastery, due }
}

export function resolveOvercharge(input: {
  battle: BattleState
  cardId: string
  exercise: Exercise
  correct: boolean
  usedHint: boolean
  responseMs: number
  srsState: SrsState
  sessionId: string
  now: number
}): { battle: BattleState; srsState: SrsState; log: Omit<RecallLog, 'id'>; outcome: OverchargeOutcome } {
  const outcome: OverchargeOutcome = !input.correct ? 'wrong' : input.usedHint ? 'correct-hint' : 'correct'
  const { state, log } = recordRecall({
    state: input.srsState,
    exercise: input.exercise,
    correct: input.correct,
    usedHint: input.usedHint,
    responseMs: input.responseMs,
    sessionId: input.sessionId,
    encounterId: `${input.sessionId}:${input.cardId}:t${input.battle.turn}`,
    now: input.now,
  })
  const battle = playCard(input.battle, input.cardId, outcome, {
    unitMastery: input.srsState.mastery,
    unitDue: input.srsState.dueAt !== undefined && input.srsState.dueAt <= input.now,
  })
  return { battle, srsState: state, log, outcome }
}
```

```ts
import type { Exercise, VocabularyUnit } from '../learning/types'
import { rngInt, rngShuffle } from '../battle/rng'
import type { RawContentPack } from './raw'
import type { ContentPack } from '../learning/types'

function fnv1a(str: string): number {
  let h = 0x811c9dc5
  for (let i = 0; i < str.length; i++) {
    h ^= str.charCodeAt(i)
    h = Math.imul(h, 0x01000193)
  }
  return h >>> 0
}

function pickDistractors(units: VocabularyUnit[], self: VocabularyUnit, seed: number, field: 'term' | 'definitionZh'): string[] {
  const samePos = units.filter(u => u.id !== self.id && u.partOfSpeech === self.partOfSpeech)
  const others = units.filter(u => u.id !== self.id && u.partOfSpeech !== self.partOfSpeech)
  const [shuffledSame] = rngShuffle(seed, samePos)
  const [shuffledOther] = rngShuffle(seed + 1, others)
  const picked = [...shuffledSame, ...shuffledOther].slice(0, 3)
  return picked.map(u => u[field])
}

function choiceExercise(unit: VocabularyUnit, units: VocabularyUnit[], kind: 'vocab-meaning-choice' | 'vocab-word-choice'): Exercise {
  const id = `${unit.id}:${kind}`
  const seed = fnv1a(id)
  const field = kind === 'vocab-meaning-choice' ? 'definitionZh' : 'term'
  const correctText = unit[field]
  const distractors = pickDistractors(units, unit, seed, field as 'term' | 'definitionZh')
  const [correctIdx] = rngInt(seed + 2, 0, 4)
  const texts: string[] = []
  let di = 0
  for (let i = 0; i < 4; i++) texts.push(i === correctIdx ? correctText : distractors[di++])
  const options = texts.map((text, i) => ({ id: `${id}:opt:${i}`, text }))
  return {
    id,
    unitId: unit.id,
    kind,
    prompt: kind === 'vocab-meaning-choice' ? unit.term : `${unit.definitionZh}${unit.partOfSpeech ? ` (${unit.partOfSpeech})` : ''}`,
    options,
    answer: { kind: 'choice', correctOptionId: `${id}:opt:${correctIdx}` },
    explanation: `${unit.term} = ${unit.definitionZh}`,
    difficulty: kind === 'vocab-meaning-choice' ? 1 : 2,
  }
}

export function expandPack(raw: RawContentPack): ContentPack {
  const vocabUnits = raw.units.filter((u): u is VocabularyUnit => u.type === 'vocab')
  const generated: Exercise[] = vocabUnits.flatMap(u => [
    choiceExercise(u, vocabUnits, 'vocab-meaning-choice'),
    choiceExercise(u, vocabUnits, 'vocab-word-choice'),
  ])
  // 无手写拼写练习的单位自动生成（导入包路径；内置包手写优先）
  const handwrittenSpelling = new Set(raw.exercises.filter(e => e.kind === 'vocab-spelling').map(e => e.unitId))
  const spellingGenerated: Exercise[] = vocabUnits
    .filter(u => !handwrittenSpelling.has(u.id))
    .map(u => ({
      id: `${u.id}:vocab-spelling`,
      unitId: u.id,
      kind: 'vocab-spelling' as const,
      prompt: `${u.definitionZh}${u.partOfSpeech ? ` (${u.partOfSpeech})` : ''}`,
      answer: { kind: 'text' as const, accepted: [u.term] },
      explanation: `${u.term} = ${u.definitionZh}`,
      difficulty: 3 as const,
    }))
  return { ...raw, exercises: [...raw.exercises, ...generated, ...spellingGenerated] }
}
```

```ts
import type { ContentPack, VocabularyUnit } from '../learning/types'

export function validatePack(pack: ContentPack): string[] {
  const problems: string[] = []
  if (pack.schemaVersion !== 1) problems.push(`unsupported schemaVersion: ${pack.schemaVersion}`)
  const ids = new Set<string>()
  for (const u of pack.units) {
    if (ids.has(u.id)) problems.push(`duplicate unit id: ${u.id}`)
    ids.add(u.id)
  }
  const exerciseIds = new Set<string>()
  for (const e of pack.exercises) {
    if (exerciseIds.has(e.id)) problems.push(`duplicate exercise id: ${e.id}`)
    exerciseIds.add(e.id)
    if (!ids.has(e.unitId)) problems.push(`exercise ${e.id} references missing unit ${e.unitId}`)
    if (e.answer.kind === 'choice') {
      const optIds = new Set(e.options?.map(o => o.id))
      if (!optIds.has(e.answer.correctOptionId)) problems.push(`exercise ${e.id}: correctOptionId not in options`)
      const texts = new Set(e.options?.map(o => o.text))
      if (texts.size !== e.options?.length) problems.push(`exercise ${e.id}: duplicate option texts`)
    }
  }
  for (const u of pack.units.filter((x): x is VocabularyUnit => x.type === 'vocab')) {
    const kinds = new Set(pack.exercises.filter(e => e.unitId === u.id).map(e => e.kind))
    for (const required of ['vocab-meaning-choice', 'vocab-word-choice', 'vocab-spelling'] as const) {
      if (!kinds.has(required)) problems.push(`unit ${u.id} missing exercise kind: ${required}`)
    }
    if (u.enrichmentStatus === 'complete' && !u.definitionZh) problems.push(`unit ${u.id} complete but empty definitionZh`)
  }
  return problems
}
```

```ts
import Papa from 'papaparse'

export interface ImportIssue { row: number; reason: string }

export interface ParsedTable {
  headers: string[]
  rows: string[][]
  delimiter: ',' | '\t'
  encodingWarning: boolean
  errors: ImportIssue[]
}

export function parseDelimited(text: string): ParsedTable {
  const encodingWarning = text.includes('�')
  const cleaned = text.replace(/^﻿/, '')
  const firstLine = cleaned.split(/\r?\n/, 1)[0] ?? ''
  const delimiter = firstLine.includes('\t') ? '\t' : ','
  const result = Papa.parse<string[]>(cleaned, { delimiter, skipEmptyLines: 'greedy' })
  const [headerRow, ...rows] = result.data as string[][]
  const errors: ImportIssue[] = result.errors.map(e => ({ row: (e.row ?? 0) + 1, reason: e.code }))
  return {
    headers: (headerRow ?? []).map(h => h.trim()),
    rows,
    delimiter,
    encodingWarning,
    errors,
  }
}
```

```ts
export function guessMapping(headers: string[]): ColumnMapping {
  const find = (aliases: string[]) => headers.find(h => aliases.includes(h.trim().toLowerCase()))

    term: find(TERM_ALIASES) ?? headers[0] ?? 'term',

    exampleZh: headers.find(h => h.toLowerCase() === 'examplezh' || h === '例句翻译'),
```

```ts
import type { ContentPack, VocabularyUnit } from '../learning/types'
import { normalizeForMatch } from '../learning/text'

export interface DictCandidate {
  unitId: string
  term: string
  definitionZh: string
  senseId: string
}

export function buildDictionaryIndex(packs: ContentPack[]): Map<string, DictCandidate[]> {
  const index = new Map<string, DictCandidate[]>()
  for (const pack of packs) {
    for (const u of pack.units) {
      if (u.type !== 'vocab') continue
      const v = u as VocabularyUnit
      if (!v.definitionZh) continue
      const key = normalizeForMatch(v.term)
      const list = index.get(key) ?? []
      list.push({ unitId: v.id, term: v.term, definitionZh: v.definitionZh, senseId: v.senseId })
      index.set(key, list)
    }
  }
  return index
}

export function lookupTerm(index: Map<string, DictCandidate[]>, normalizedTerm: string): DictCandidate[] {
  return index.get(normalizedTerm) ?? []
}
```

`src/core/import/build.ts`:

```ts
import { normalizeDisplay, normalizeForMatch } from '../learning/text'
import type { VocabularyUnit } from '../learning/types'
import type { DictCandidate } from './enrich'
import { lookupTerm } from './enrich'
import type { ColumnMapping } from './mapping'
import type { ImportIssue, ParsedTable } from './parse'

export interface ImportBuildResult {
  units: VocabularyUnit[]
  pending: Array<{ row: number; term: string }>
  needsChoice: Array<{ row: number; term: string; candidates: DictCandidate[] }>
  duplicates: ImportIssue[]
  errors: ImportIssue[]
}

export function buildImportedUnits(
  table: ParsedTable,
  mapping: ColumnMapping,
  ctx: {
    now: number
    sourceName: string
    dictionary: Map<string, DictCandidate[]>
    userChoices: Map<number, string>     // row -> senseId
    userDefinitions: Map<number, string> // row -> definition
  },
): ImportBuildResult {
  const col = (row: string[], name?: string) => {
    if (!name) return undefined
    const idx = table.headers.indexOf(name)
    const v = idx >= 0 ? row[idx] : undefined
    return v !== undefined && v.trim() !== '' ? normalizeDisplay(v) : undefined
  }
  const units: VocabularyUnit[] = []
  const pending: ImportBuildResult['pending'] = []
  const needsChoice: ImportBuildResult['needsChoice'] = []
  const duplicates: ImportIssue[] = []
  const errors: ImportIssue[] = [...table.errors]
  const seen = new Set<string>()

  table.rows.forEach((row, i) => {
    const rowNumber = i + 2 // 表头为第 1 行
    const rawTerm = col(row, mapping.term)
    if (!rawTerm) {
      errors.push({ row: rowNumber, reason: 'missing-term' })
      return
    }
    const term = rawTerm
    const normalizedTerm = normalizeForMatch(term)
    let definitionZh = col(row, mapping.definition) ?? ctx.userDefinitions.get(rowNumber)
    let senseId = `${normalizedTerm}:import`
    let sourceUnitId: string | undefined

    if (!definitionZh) {
      const candidates = lookupTerm(ctx.dictionary, normalizedTerm)
      const chosen = ctx.userChoices.get(rowNumber)
      if (chosen) {
        const c = candidates.find(x => x.senseId === chosen)
        if (c) {
          definitionZh = c.definitionZh
          senseId = c.senseId
          sourceUnitId = c.unitId
        }
      } else if (candidates.length === 1) {
        definitionZh = candidates[0].definitionZh
        senseId = candidates[0].senseId
        sourceUnitId = candidates[0].unitId
      } else if (candidates.length > 1) {
        needsChoice.push({ row: rowNumber, term, candidates })
        return
      }
    }

    const dedupKey = `${normalizedTerm}::${normalizeForMatch(definitionZh ?? '')}`
    if (seen.has(dedupKey)) {
      duplicates.push({ row: rowNumber, reason: `duplicate: ${term}` })
      return
    }
    seen.add(dedupKey)

    if (!definitionZh) {
      pending.push({ row: rowNumber, term })
      return
    }

    units.push({
      id: `import:${ctx.sourceName}:${rowNumber}`,
      type: 'vocab',
      term,
      normalizedTerm,
      senseId,
      definitionZh,
      phonetic: col(row, mapping.phonetic),
      partOfSpeech: col(row, mapping.partOfSpeech),
      example: col(row, mapping.example),
      exampleZh: col(row, mapping.exampleZh),
      curriculumTags: ['custom'],
      enrichmentStatus: 'complete',
      source: { kind: 'import', name: ctx.sourceName, importedAt: ctx.now },
    })
    void sourceUnitId
  })

  return { units, pending, needsChoice, duplicates, errors }
}
```

```ts
import { openDB, type IDBPDatabase } from 'idb'
import type { ContentPack } from '../../core/learning/types'
import type { RecallLog, SrsState } from '../../core/srs/state'
import type { SessionState } from '../../core/session/types'

export const DB_SCHEMA_VERSION = 1
const DB_NAME = 'english-roguelike'

let dbPromise: Promise<IDBPDatabase> | null = null

function db(): Promise<IDBPDatabase> {
  if (!dbPromise) {
    dbPromise = openDB(DB_NAME, DB_SCHEMA_VERSION, {
      upgrade(d) {
        d.createObjectStore('packs', { keyPath: 'id' })
        d.createObjectStore('srs', { keyPath: 'unitId' })
        const logs = d.createObjectStore('logs', { keyPath: 'id' })
        logs.createIndex('by-session', 'sessionId')
        logs.createIndex('by-unit', 'unitId')
        d.createObjectStore('settings', { keyPath: 'key' })
        d.createObjectStore('snapshots', { keyPath: 'sessionId' })
      },
    })
  }
  return dbPromise
}

export async function savePack(pack: ContentPack): Promise<void> {
  await (await db()).put('packs', pack)
}

export async function loadAllPacks(): Promise<ContentPack[]> {
  return (await db()).getAll('packs')
}

export async function loadSrsStates(unitIds: string[]): Promise<Map<string, SrsState>> {
  const d = await db()
  const map = new Map<string, SrsState>()
  for (const id of unitIds) {
    const s = await d.get('srs', id)
    if (s) map.set(id, s as SrsState)
  }
  return map
}

export async function loadAllSrsStates(): Promise<Map<string, SrsState>> {
  const all = await (await db()).getAll('srs')
  return new Map(all.map((s: SrsState) => [s.unitId, s]))
}

/** SRS 状态更新与计分日志同一事务（spec 12.3） */
export async function saveSrsAndLog(state: SrsState, log: Omit<RecallLog, 'id'>): Promise<void> {
  const d = await db()
  const tx = d.transaction(['srs', 'logs'], 'readwrite')
  await tx.objectStore('srs').put(state)
  await tx.objectStore('logs').put({ ...log, id: crypto.randomUUID() })
  await tx.done
}

export async function appendTeachingLogs(logs: Array<Omit<RecallLog, 'id'>>): Promise<void> {
  const d = await db()
  const tx = d.transaction('logs', 'readwrite')
  for (const log of logs) await tx.objectStore('logs').put({ ...log, id: crypto.randomUUID() })
  await tx.done
}

export async function loadLogsBySession(sessionId: string): Promise<RecallLog[]> {
  return (await db()).getAllFromIndex('logs', 'by-session', sessionId)
}

export async function saveSnapshot(s: SessionState): Promise<void> {
  await (await db()).put('snapshots', { ...s, sessionId: s.id })
}

export async function loadLatestSnapshot(): Promise<SessionState | undefined> {
  const all = await (await db()).getAll('snapshots')
  if (all.length === 0) return undefined
  all.sort((a, b) => b.createdAt - a.createdAt)
  return all[0] as SessionState
}

export async function deleteSnapshot(sessionId: string): Promise<void> {
  await (await db()).delete('snapshots', sessionId)
}

export async function saveSettings(key: string, value: unknown): Promise<void> {
  await (await db()).put('settings', { key, value })
}

export async function loadSettings<T>(key: string): Promise<T | undefined> {
  const row = await (await db()).get('settings', key)

  ])
```

```ts
export function detectSpeechSupport(): { ok: boolean; reason?: string } {
  if (typeof window === 'undefined' || !('speechSynthesis' in window)) {
    return { ok: false, reason: 'speechSynthesis unavailable' }
  }
  return { ok: true }
}

export function speakEnglish(text: string): Promise<void> {
  return new Promise((resolve, reject) => {
    if (!detectSpeechSupport().ok) {
      reject(new Error('speech unavailable'))
      return
    }
    const utter = new SpeechSynthesisUtterance(text)
    utter.lang = 'en-US'
    utter.rate = 0.9
    utter.onend = () => resolve()
    utter.onerror = e => reject(new Error(e.error))
    window.speechSynthesis.cancel()
    window.speechSynthesis.speak(utter)
  })
}
```

```ts
import { create } from 'zustand'
import { BALANCE } from '../../core/battle/balance'
import { BATTLE_ORDER, ENEMIES, STARTER_TEMPLATES } from '../../core/battle/data'
import { createBattle, endTurn as engineEndTurn, playCard, type BattleState } from '../../core/battle/engine'
import { loadBuiltinPacks } from '../../core/content/load'
import { checkAnswer } from '../../core/learning/answer'
import type { ContentPack, Exercise, VocabularyUnit } from '../../core/learning/types'
import { prepareOvercharge, resolveOvercharge } from '../../core/session/overcharge'
import { computeSettlement, createSession, phaseAfterBattle } from '../../core/session/session'
import type { SessionState, SettlementSummary } from '../../core/session/types'
import { initialSrsState, type SrsState } from '../../core/srs/state'
import { completeTeaching } from '../../core/srs/teaching'
import {
  deleteSnapshot, loadAllPacks, loadAllSrsStates, loadLatestSnapshot,
  loadSrsStates, saveSrsAndLog, saveSnapshot, clearAllData,
} from '../../infra/indexeddb/db'
import { detectSpeechSupport } from '../../infra/speech/speech'

export type Screen = 'home' | 'teaching' | 'battle' | 'reward-relic' | 'reward-heal'
  | 'settlement' | 'defeat' | 'import' | 'settings' | 'progress'

export interface OverchargeState {
  cardId: string
  unitId: string
  exercise: Exercise
  startedAt: number
  usedHint: boolean
  pendingConfirm?: string
}

export interface FeedbackState {
  correct: boolean
  answerText: string
  explanation: string
}

const templates = Object.fromEntries(STARTER_TEMPLATES.map(t => [t.id, t]))

interface GameStore {
  screen: Screen
  packs: ContentPack[]
  pack: ContentPack | null
  srsStates: Map<string, SrsState>
  session: SessionState | null
  battle: BattleState | null
  selectedCardId: string | null
  overcharge: OverchargeState | null
  feedback: FeedbackState | null
  settlement: SettlementSummary | null
  recentExerciseIds: string[]
  speechOk: boolean
  hasSnapshot: boolean

  init: () => Promise<void>
  setScreen: (s: Screen) => void
  startSession: (packId: string, newQuota: number) => Promise<void>
  resumeSession: () => Promise<void>
  teachNext: () => Promise<void>
  beginBattle: (index: number) => void
  selectCard: (cardId: string | null) => void
  playSelectedDirect: () => void
  startOvercharge: () => void
  cancelOvercharge: () => void
  useHint: () => void
  submitOvercharge: (response: string | string[]) => Promise<void>
  confirmNearMiss: (submitAsIs: boolean) => Promise<void>
  acknowledgeFeedback: () => void
  endTurnAction: () => void
  chooseRelic: (relicId: string) => Promise<void>
  claimHeal: () => Promise<void>
  backHome: () => void
  resetAll: () => Promise<void>
}

export const useGameStore = create<GameStore>((set, get) => {
  const now = () => Date.now()

  const persistSnapshot = async () => {
    const { session } = get()
    if (session) await saveSnapshot(session)
  }

  const routeAfterBattle = async (battle: BattleState) => {
    const { session } = get()
    if (!session || !battle.result) return
    const phase = phaseAfterBattle(session, battle.result)
    const updated: SessionState = { ...session, phase, playerHp: Math.max(0, battle.player.hp) }
    set({ session: updated })
    await persistSnapshot()
    if (phase.kind === 'settlement') {
      const { pack, srsStates } = get()
      const terms = new Map((pack?.units ?? []).filter((u): u is VocabularyUnit => u.type === 'vocab')
        .map(u => [u.id, { term: u.term, definitionZh: u.definitionZh }]))
      set({ settlement: computeSettlement(updated, srsStates, terms), screen: 'settlement' })
      await deleteSnapshot(updated.id)
    } else if (phase.kind === 'defeat') {
      set({ screen: 'defeat' })
      await deleteSnapshot(updated.id)
    } else if (phase.kind === 'reward-relic') {
      set({ screen: 'reward-relic' })
    } else if (phase.kind === 'reward-heal') {
      set({ screen: 'reward-heal' })
    }
  }

  return {
    screen: 'home',
    packs: [],
    pack: null,
    srsStates: new Map(),
    session: null,
    battle: null,
    selectedCardId: null,
    overcharge: null,
    feedback: null,
    settlement: null,
    recentExerciseIds: [],
    speechOk: false,
    hasSnapshot: false,

    init: async () => {
      const builtin = loadBuiltinPacks()
      const custom = await loadAllPacks()
      const snapshot = await loadLatestSnapshot()
      set({ packs: [...builtin, ...custom], speechOk: detectSpeechSupport().ok, hasSnapshot: !!snapshot })
    },

    setScreen: screen => set({ screen }),

    startSession: async (packId, newQuota) => {
      const pack = get().packs.find(p => p.id === packId)
      if (!pack) return
      const unitIds = pack.units.map(u => u.id)
      const srsStates = await loadSrsStates(unitIds)
      const session = createSession({
        pack, srsStates, now: now(), newQuota,
        sessionId: crypto.randomUUID(), seed: Math.floor(Math.random() * 2 ** 31),
      })
      set({ pack, srsStates, session, battle: null, settlement: null, recentExerciseIds: [], feedback: null })
      await persistSnapshot()
      if (session.phase.kind === 'teaching') {
        set({ screen: 'teaching' })
      } else {
        get().beginBattle(0)
      }
    },

    resumeSession: async () => {
      const snapshot = await loadLatestSnapshot()
      if (!snapshot) return
      const pack = get().packs.find(p => p.id === snapshot.packId)
      if (!pack) return
      const srsStates = await loadSrsStates(pack.units.map(u => u.id))
      set({ pack, session: snapshot, srsStates })
      if (snapshot.phase.kind === 'teaching') set({ screen: 'teaching' })
      else if (snapshot.phase.kind === 'battle') get().beginBattle(snapshot.battleIndex)
      else if (snapshot.phase.kind === 'reward-relic') set({ screen: 'reward-relic' })
      else if (snapshot.phase.kind === 'reward-heal') set({ screen: 'reward-heal' })
      else set({ screen: 'home' })
    },

    teachNext: async () => {
      const { session, srsStates } = get()
      if (!session || session.phase.kind !== 'teaching') return
      const [unitId, ...rest] = session.phase.pendingUnitIds
      const current = srsStates.get(unitId) ?? initialSrsState(unitId)
      const { state, log } = completeTeaching(current, now())
      await saveSrsAndLog(state, { ...log, sessionId: session.id })
      const nextStates = new Map(srsStates)
      nextStates.set(unitId, state)
      const phase: SessionState['phase'] = rest.length > 0
        ? { kind: 'teaching', pendingUnitIds: rest }
        : { kind: 'battle', battleIndex: 0 }
      set({ srsStates: nextStates, session: { ...session, phase } })
      await persistSnapshot()
      if (rest.length === 0) get().beginBattle(0)
    },

    beginBattle: index => {
      const { session } = get()
      if (!session) return
      const enemyDef = ENEMIES[BATTLE_ORDER[index]]
      const battle = createBattle({
        seed: session.seed * 31 + index,
        playerHp: session.playerHp,
        playerMaxHp: session.playerMaxHp,
        deck: session.deck,
        enemyDef,
        relics: session.relics,
        templates,
      })
      set({ battle, selectedCardId: null, overcharge: null, feedback: null, screen: 'battle',
        session: { ...session, battleIndex: index, phase: { kind: 'battle', battleIndex: index } } })
      void persistSnapshot()
    },

    selectCard: cardId => set({ selectedCardId: cardId }),

    playSelectedDirect: () => {
      const { battle, selectedCardId } = get()
      if (!battle || !selectedCardId) return
      const next = playCard(battle, selectedCardId, null, {})
      set({ battle: next, selectedCardId: null })
      if (next.result) void routeAfterBattle(next)
    },

    startOvercharge: () => {
      const { battle, selectedCardId, pack, srsStates, speechOk, recentExerciseIds } = get()
      if (!battle || !selectedCardId || !pack) return
      const r = prepareOvercharge({
        battle, cardId: selectedCardId, pack, srsStates,
        speechAvailable: speechOk, recentExerciseIds, now: now(),
      })
      if ('error' in r) return
      set({ overcharge: { cardId: selectedCardId, unitId: r.unitId, exercise: r.exercise, startedAt: now(), usedHint: false } })
    },

    cancelOvercharge: () => set({ overcharge: null }),

    useHint: () => {
      const { overcharge } = get()
      if (overcharge) set({ overcharge: { ...overcharge, usedHint: true } })
    },

    submitOvercharge: async response => {
      const { overcharge, battle, srsStates, session } = get()
      if (!overcharge || !battle || !session) return
      const result = checkAnswer(overcharge.exercise, response)
      if (result.suggestConfirm && typeof response === 'string') {
        set({ overcharge: { ...overcharge, pendingConfirm: response } })
        return
      }
      await finishOvercharge(result.correct)
    },

    confirmNearMiss: async submitAsIs => {
      const { overcharge } = get()
      if (!overcharge?.pendingConfirm) return
      if (!submitAsIs) {
        set({ overcharge: { ...overcharge, pendingConfirm: undefined } })
        return
      }
      await finishOvercharge(false) // 近似拼写按提交判错（spec：不直接判对）
    },

    acknowledgeFeedback: () => {
      const { battle } = get()
      set({ feedback: null, selectedCardId: null })
      if (battle?.result) void routeAfterBattle(battle)
    },

    endTurnAction: () => {
      const { battle } = get()
      if (!battle) return
      const next = engineEndTurn(battle)
      set({ battle: next, selectedCardId: null })
      if (next.result) void routeAfterBattle(next)
    },

    chooseRelic: async relicId => {
      const { session } = get()
      if (!session) return
      const updated = { ...session, relics: [...session.relics, relicId] }
      set({ session: updated })
      await persistSnapshot()
      get().beginBattle(1)
    },

    claimHeal: async () => {
      const { session } = get()
      if (!session) return
      const updated = { ...session, playerHp: Math.min(session.playerMaxHp, session.playerHp + BALANCE.intermissionHeal) }
      set({ session: updated })
      await persistSnapshot()
      get().beginBattle(2)
    },

    backHome: () => set({ screen: 'home', session: null, battle: null, settlement: null }),

    resetAll: async () => {
      await clearAllData()
      set({ srsStates: new Map(), session: null, battle: null, settlement: null, hasSnapshot: false, screen: 'home' })
      await get().init()
    },
  }

  async function finishOvercharge(correct: boolean) {
    const { overcharge, battle, srsStates, session } = get()
    if (!overcharge || !battle || !session) return
    const srsState = srsStates.get(overcharge.unitId) ?? initialSrsState(overcharge.unitId)
    const r = resolveOvercharge({
      battle, cardId: overcharge.cardId, exercise: overcharge.exercise,
      correct, usedHint: overcharge.usedHint,
      responseMs: now() - overcharge.startedAt,
      srsState, sessionId: session.id, now: now(),
    })
    await saveSrsAndLog(r.srsState, r.log)
    const nextStates = new Map(srsStates)
    nextStates.set(overcharge.unitId, r.srsState)
    const answerText = overcharge.exercise.answer.kind === 'choice'
      ? overcharge.exercise.options?.find(o => o.id === (overcharge.exercise.answer as { correctOptionId: string }).correctOptionId)?.text ?? ''
      : (overcharge.exercise.answer as { accepted: string[] }).accepted[0]
    set({
      battle: r.battle,
      srsStates: nextStates,
      overcharge: null,
      recentExerciseIds: [...get().recentExerciseIds, overcharge.exercise.id],
      session: { ...session, attempts: [...session.attempts, { unitId: overcharge.unitId, correct, graded: r.log.graded, kind: overcharge.exercise.kind, at: now() }] },
      feedback: { correct, answerText, explanation: overcharge.exercise.explanation },
    })
    await persistSnapshot()
  }
})
```

```ts
import { useEffect } from 'react'

export interface BattleKeyActions {
  selectByIndex: (i: number) => void
  playDirect: () => void
  overcharge: () => void
  endTurn: () => void
  cancel: () => void
}

export function useBattleKeyboard(actions: BattleKeyActions, enabled: boolean): void {
  useEffect(() => {
    if (!enabled) return
    const onKey = (e: KeyboardEvent) => {
      if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return
      if (/^[1-9]$/.test(e.key)) actions.selectByIndex(Number(e.key) - 1)
      else if (e.key === 'Enter') actions.playDirect()
      else if (e.key === 'o' || e.key === 'O') actions.overcharge()
      else if (e.key === 'e' || e.key === 'E') actions.endTurn()
      else if (e.key === 'Escape') actions.cancel()
    }
    window.addEventListener('keydown', onKey)
    return () => window.removeEventListener('keydown', onKey)
  }, [actions, enabled])
}
```

```ts
import type { CardTemplate, Intent } from '../../core/battle/types'

export function describeEffects(tpl: CardTemplate): string {
  return tpl.effects.map(fx => {
    switch (fx.type) {
      case 'damage': return `造成 ${fx.amount} 点伤害`
      case 'block': return `获得 ${fx.amount} 点格挡`
      case 'draw': return `抽 ${fx.amount} 张牌`
      case 'apply-status': return `给予敌人 ${fx.stacks} 回合虚弱`
    }
  }).join('；')
}

export function describeIntent(intent: Intent): string {
  switch (intent.type) {
    case 'attack': return intent.hits && intent.hits > 1 ? `⚔ 攻击 ${intent.amount}×${intent.hits}` : `⚔ 攻击 ${intent.amount}`
    case 'attack-block': return `⚔ 攻击 ${intent.amount} + 🛡 格挡 ${intent.block}`
    case 'block-debuff': return `🛡 格挡 ${intent.block} + 虚弱 ${intent.stacks}`
  }
}
```

```tsx
import { useEffect, useRef, useState } from 'react'
import { BALANCE } from '../../core/battle/balance'
import type { VocabularyUnit } from '../../core/learning/types'
import { speakEnglish } from '../../infra/speech/speech'
import { useGameStore } from '../state/game-store'
import { useSettingsStore } from '../state/settings-store'

const SUGGESTED: Record<string, number> = {
  'vocab-meaning-choice': BALANCE.suggestedTimeMs.choice,
  'vocab-word-choice': BALANCE.suggestedTimeMs.wordChoice,
  'vocab-spelling': BALANCE.suggestedTimeMs.spelling,
  'vocab-listening-spelling': BALANCE.suggestedTimeMs.listening,
}

export function OverchargePanel() {
  const { overcharge, pack, submitOvercharge, confirmNearMiss, useHint, cancelOvercharge, speechOk } = useGameStore()
  const { timerHints, speechEnabled } = useSettingsStore()
  const [text, setText] = useState('')
  const [elapsed, setElapsed] = useState(0)
  const inputRef = useRef<HTMLInputElement>(null)

  const suggested = SUGGESTED[overcharge?.exercise.kind ?? ''] ?? 8000
  useEffect(() => {
    if (!timerHints) return
    const t = setInterval(() => setElapsed(Date.now() - (overcharge?.startedAt ?? Date.now())), 200)
    return () => clearInterval(t)
  }, [timerHints, overcharge?.startedAt])
  useEffect(() => { inputRef.current?.focus() }, [overcharge?.exercise.id])

  if (!overcharge) return null
  const { exercise } = overcharge
  const unit = pack?.units.find((u): u is VocabularyUnit => u.id === overcharge.unitId && u.type === 'vocab')
  const isChoice = exercise.answer.kind === 'choice'
  const isListening = exercise.kind === 'vocab-listening-spelling'

  return (
    <div className="dock" role="dialog" aria-label="过载回忆">
      <div className="dock-inner">
        {timerHints && (
          <div className="timer-track" aria-hidden="true">
            <div style={{ width: `${Math.min(100, (elapsed / suggested) * 100)}%` }} />
          </div>
        )}
        <div className="row" style={{ justifyContent: 'space-between' }}>
          <strong>过载：{unit?.term ?? ''}</strong>
          <span className="muted">超时不判负，放心作答</span>

            <p>你的拼写「{overcharge.pendingConfirm}」接近但可能有误。确定提交吗？</p>
            <div className="row">
              <button autoFocus onClick={() => setText('') || void confirmNearMiss(false)}>重新输入</button>
              <button onClick={() => void confirmNearMiss(true)}>仍要提交</button>
            </div>
          </div>
        ) : (
          <>
            <p style={{ fontSize: 18 }}>
              {isListening ? (
                <button aria-label="播放发音" onClick={() => unit && void speakEnglish(unit.term)}
                  disabled={!speechOk || !speechEnabled}>🔊 播放发音，听词拼写</button>
              ) : exercise.prompt}
            </p>
            {isChoice && exercise.options && (
              <div className="choice-grid">
                {exercise.options.map(o => (
                  <button key={o.id} onClick={() => void submitOvercharge(o.id)}>{o.text}</button>

              </button>
              <button onClick={cancelOvercharge}>取消（Esc）</button>
            </div>
          </>

  )
```

```ts
/** 错误报告 CSV 下载；转义可能被表格软件解释为公式的单元格（spec 8.3） */
export function downloadErrorReport(issues: ImportIssue[], filename: string): void {
  const esc = (s: string) => (/^[=+\-@]/.test(s) ? `'${s}` : s).replace(/"/g, '""')
  const csv = ['row,reason', ...issues.map(i => `${i.row},"${esc(i.reason)}"`)].join('\n')
  const url = URL.createObjectURL(new Blob([csv], { type: 'text/csv' }))
  const a = document.createElement('a')
  a.href = url
  a.download = filename
  a.click()
  URL.revokeObjectURL(url)
}
```

```tsx
import { useMemo, useState } from 'react'
import { BALANCE } from '../../core/battle/balance'
import { expandPack } from '../../core/content/expand'
import { validatePack } from '../../core/content/validate'
import { buildImportedUnits, type ImportBuildResult } from '../../core/import/build'
import { buildDictionaryIndex, type DictCandidate } from '../../core/import/enrich'
import { guessMapping, type ColumnMapping } from '../../core/import/mapping'
import { parseDelimited, type ParsedTable } from '../../core/import/parse'
import { savePack } from '../../infra/indexeddb/db'
import { useGameStore } from '../state/game-store'
import { downloadErrorReport } from './report'

type Step = 'source' | 'mapping' | 'enrich' | 'done'

export function ImportWizard() {
  const { packs, setScreen, init } = useGameStore()
  const [step, setStep] = useState<Step>('source')
  const [sourceName, setSourceName] = useState('paste')
  const [table, setTable] = useState<ParsedTable | null>(null)
  const [mapping, setMapping] = useState<ColumnMapping>({ term: '' })
  const [result, setResult] = useState<ImportBuildResult | null>(null)
  const [choices, setChoices] = useState<Map<number, string>>(new Map())
  const [definitions, setDefinitions] = useState<Map<number, string>>(new Map())
  const [savedCount, setSavedCount] = useState(0)
  const [fileError, setFileError] = useState('')

  const dictionary = useMemo(() => buildDictionaryIndex(packs), [packs])

  const loadText = (text: string, name: string) => {
    if (new Blob([text]).size > BALANCE.importMaxBytes) {
      setFileError(`文件超过 ${BALANCE.importMaxBytes / 1024 / 1024} MB 限制`)
      return
    }
    const t = parseDelimited(text)
    if (t.rows.length > BALANCE.importMaxRows) {
      setFileError(`行数超过 ${BALANCE.importMaxRows} 限制`)
      return
    }
    setFileError('')
    setSourceName(name)
    setTable(t)
    setMapping(guessMapping(t.headers))
    setStep('mapping')
  }

  const runBuild = (c: Map<number, string>, d: Map<number, string>) => {
    if (!table) return
    setResult(buildImportedUnits(table, mapping, { now: Date.now(), sourceName, dictionary, userChoices: c, userDefinitions: d }))
  }

  const commit = async () => {
    if (!table || !result) return
    const pack = expandPack({
      id: `custom:${Date.now()}`,
      schemaVersion: 1,
      name: `导入：${sourceName}`,
      stage: 'custom',
      curriculumTags: ['custom'],
      units: result.units,
      exercises: [],
      source: { kind: 'import', name: sourceName, importedAt: Date.now() },
    })
    const problems = validatePack(pack)
    if (problems.length > 0) {
      setFileError(`校验失败：${problems[0]}`)
      return
    }
    await savePack(pack)
    await init()
    setSavedCount(result.units.length)
    setStep('done')
  }

  return (
    <div className="screen">
      <h2>导入自定义词表</h2>
      {fileError && <p className="feedback-bad" role="alert">{fileError}</p>}

      {step === 'source' && (
        <div className="panel">
          <p>支持 CSV / TSV（UTF-8，首行表头，必填 term 列）。<a href="data:text/csv,term,definition,example%0Aapple,苹果,I eat an apple." download="template.csv">下载模板</a></p>
          <input type="file" accept=".csv,.tsv,.txt" aria-label="选择词表文件"
            onChange={async e => {
              const f = e.target.files?.[0]
              if (f) loadText(await f.text(), f.name)
            }} />
          <p className="muted">或粘贴表格文本：</p>
          <textarea rows={6} style={{ width: '100%' }} id="paste-area" aria-label="粘贴表格文本" />
          <button onClick={() => loadText((document.getElementById('paste-area') as HTMLTextAreaElement).value, 'paste')}>解析</button>
          <button onClick={() => setScreen('home')}>取消</button>
        </div>
      )}

      {step === 'mapping' && table && (
        <div className="panel">
          {table.encodingWarning && <p role="alert" className="feedback-bad">检测到乱码字符：文件可能不是 UTF-8 编码，请转换后重试。</p>}
          <p className="muted">分隔符：{table.delimiter === '\t' ? '制表符' : '逗号'} · {table.rows.length} 行</p>
          <div className="row">
            <label>term 列：
              <select value={mapping.term} onChange={e => setMapping({ ...mapping, term: e.target.value })}>
                {table.headers.map(h => <option key={h} value={h}>{h}</option>)}
              </select>
            </label>
            <label>definition 列：
              <select value={mapping.definition ?? ''} onChange={e => setMapping({ ...mapping, definition: e.target.value || undefined })}>
                <option value="">（无）</option>
                {table.headers.map(h => <option key={h} value={h}>{h}</option>)}
              </select>
            </label>
          </div>
          <table className="preview">
            <thead><tr><th>#</th>{table.headers.map(h => <th key={h}>{h}</th>)}</tr></thead>
            <tbody>
              {table.rows.slice(0, 10).map((r, i) => (
                <tr key={i}><td>{i + 2}</td>{table.headers.map((_, j) => <td key={j}>{r[j]}</td>)}</tr>
              ))}
            </tbody>
          </table>
          {table.errors.length > 0 && <p className="feedback-bad">{table.errors.length} 行解析错误（提交后可在报告中下载）</p>}
          <div className="row">
            <button onClick={() => setStep('source')}>上一步</button>
            <button autoFocus onClick={() => { runBuild(new Map(), new Map()); setStep('enrich') }}>下一步</button>
          </div>
        </div>
      )}

      {step === 'enrich' && result && (
        <div className="panel">
          <p>可导入 {result.units.length} 项 · 重复 {result.duplicates.length} · 错误 {result.errors.length} · 待补释义 {result.pending.length} · 待选词义 {result.needsChoice.length}</p>
          {result.needsChoice.map(nc => (
            <div key={nc.row} className="panel">
              <strong>第 {nc.row} 行「{nc.term}」有多个词义，请选择一个：</strong>
              {nc.candidates.map((c: DictCandidate) => (
                <label key={c.senseId} className="row">
                  <input type="radio" name={`choice-${nc.row}`} checked={choices.get(nc.row) === c.senseId}
                    onChange={() => setChoices(new Map(choices).set(nc.row, c.senseId))} />
                  {c.definitionZh}
                </label>
              ))}
            </div>
          ))}
          {result.pending.map(p => (
            <div key={p.row} className="row">
              <span>第 {p.row} 行「{p.term}」缺释义：</span>
              <input type="text" aria-label={`为 ${p.term} 补充释义`}
                onChange={e => setDefinitions(new Map(definitions).set(p.row, e.target.value))} />
            </div>
          ))}
          <div className="row">
            <button onClick={() => setStep('mapping')}>上一步</button>
            <button onClick={() => runBuild(choices, definitions)}>应用修改</button>
            <button autoFocus disabled={result.needsChoice.length > 0 || result.units.length === 0} onClick={() => void commit()}>
              确认导入 {result.units.length} 项
            </button>
          </div>
        </div>
      )}

      {step === 'done' && result && (
        <div className="panel" style={{ textAlign: 'center' }}>
          <h2>导入完成</h2>
          <p>新增 {savedCount} 个学习项目{result.pending.length > 0 && `；${result.pending.length} 项缺释义未导入`}</p>
          {(result.errors.length > 0 || result.duplicates.length > 0) && (
            <button onClick={() => downloadErrorReport([...result.errors, ...result.duplicates], 'import-report.csv')}>
              下载错误报告（{result.errors.length + result.duplicates.length} 行）
            </button>
          )}
          <div className="row" style={{ justifyContent: 'center' }}>
            <button autoFocus onClick={() => setScreen('home')}>返回主页</button>
          </div>
        </div>
      )}
    </div>
  )
}
```

注意：导入包只有 units 没有手写拼写练习——`expandPack` 已为这类单位自动生成拼写练习（见 Task 8 实现与「无手写拼写时自动生成」测试用例），因此校验可通过。

```ts
import { describe, it, expect } from 'vitest'
import { BATTLE_ORDER, ENEMIES, STARTER_TEMPLATES } from './battle/data'
import { createBattle, endTurn, playCard } from './battle/engine'

import { parseDelimited } from './import/parse'
import { prepareOvercharge, resolveOvercharge } from './session/overcharge'
import { createSession } from './session/session'
import { buildLearningSet } from './srs/scheduler'
import { initialSrsState, DAY_MS, type RecallLog, type SrsState } from './srs/state'
import { completeTeaching } from './srs/teaching'
import { recordRecall } from './srs/record'
import type { Exercise } from './learning/types'

const NOW = 1_800_000_000_000
const templates = Object.fromEntries(STARTER_TEMPLATES.map(t => [t.id, t]))
const choiceEx: Exercise = {
  id: 'e1', unitId: 'u1', kind: 'vocab-meaning-choice', prompt: 'abandon',
  options: [{ id: 'o1', text: '放弃；抛弃' }, { id: 'o2', text: '能力' }],
  answer: { kind: 'choice', correctOptionId: 'o1' }, explanation: '', difficulty: 1,

describe('验收：重新开局不能让同一 dueAt 再次计分（spec 12.4）', () => {
  it('新 sessionId 同一到期窗口仍不计分', () => {
    const taught = completeTeaching(initialSrsState('u1'), NOW).state
    const due = NOW + DAY_MS
    const r1 = recordRecall({ state: taught, exercise: choiceEx, correct: true, usedHint: false, sessionId: 'S1', now: due })
    expect(r1.log.graded).toBe(true)
    const r2 = recordRecall({ state: r1.state, exercise: choiceEx, correct: true, usedHint: false, sessionId: 'S2-新局', now: due + 1000 })
    expect(r2.log.graded).toBe(false)
    expect(r2.state).toEqual(r1.state)
  })
})

describe('验收：遗物不改变 SRS 计分结果（spec 12.4）', () => {
  it('同一次回忆，有无遗物的 log 与 srsState 完全一致', () => {
    const [pack] = loadBuiltinPacks()
    const unit = pack.units[0]
    const taught = completeTeaching(initialSrsState(unit.id), NOW).state
    const due = NOW + DAY_MS
    const deck = [{ id: 'c1', templateId: 'strike', learningUnitId: unit.id, sessionId: 'S1', upgraded: false }]
    const mk = (relics: string[]) => createBattle({

    })
    const exercise = pack.exercises.find(e => e.unitId === unit.id && e.kind === 'vocab-meaning-choice')!
    const a = resolveOvercharge({ battle: mk([]), cardId: 'c1', exercise, correct: true, usedHint: false, responseMs: 900, srsState: taught, sessionId: 'S1', now: due })
    const b = resolveOvercharge({ battle: mk(['root-compass', 'error-notebook', 'fluency-charm']), cardId: 'c1', exercise, correct: true, usedHint: false, responseMs: 900, srsState: taught, sessionId: 'S1', now: due })

    expect(a.srsState).toEqual(b.srsState)
  })

describe('验收：导入缺释义不能进战斗，补全后可启用（spec 12.4）', () => {
  it('pending 被调度器排除；userDefinitions 补全后进入候选', () => {

    const noDef = buildImportedUnits(table, { term: 'term', definition: 'definition' }, { now: NOW, sourceName: 't', dictionary: new Map(), userChoices: new Map(), userDefinitions: new Map() })

describe('验收：Web Speech 不可用仍可完成整局（spec 12.4）', () => {
  it('全流程模拟：教学 → 三战胜利 → 无听音题、有计分日志、直接出牌零日志', () => {
    const [pack] = loadBuiltinPacks()
    const states = new Map<string, SrsState>()
    // 预置 10 个已教学且已到期的单位
    for (const u of pack.units.slice(0, 10)) {
      const t = completeTeaching(initialSrsState(u.id), NOW - 2 * DAY_MS)
      states.set(u.id, { ...t.state, dueAt: NOW - DAY_MS })
    }
    const session = createSession({ pack, srsStates: states, now: NOW, newQuota: 4, sessionId: 'SIM', seed: 7 })
    expect(session.learningSet.due.length).toBe(8)
    expect(session.learningSet.fresh.length).toBe(4)

    const logs: Array<Omit<RecallLog, 'id'>> = []
    for (const unitId of session.learningSet.fresh) {
      const { state, log } = completeTeaching(states.get(unitId) ?? initialSrsState(unitId), NOW)
      states.set(unitId, state)
      logs.push({ ...log, sessionId: session.id })
    }
    const logsBeforeBattle = logs.length

    let playerHp = session.playerMaxHp
    let relics: string[] = []
    const recent: string[] = []
    for (let battleIndex = 0; battleIndex < 3; battleIndex++) {
      let battle = createBattle({
        seed: session.seed * 31 + battleIndex, playerHp, playerMaxHp: session.playerMaxHp,
        deck: session.deck, enemyDef: ENEMIES[BATTLE_ORDER[battleIndex]], relics, templates,
      })
      let guard = 0
      while (!battle.result && guard++ < 200) {
        for (const cardId of [...battle.hand]) {
          if (battle.result) break
          const card = battle.cards[cardId]
          const tpl = battle.templates[card.templateId]
          if (battle.energy < tpl.cost) continue
          if (card.learningUnitId && battle.focus > 0) {
            const prep = prepareOvercharge({ battle, cardId, pack, srsStates: states, speechAvailable: false, recentExerciseIds: recent, now: NOW })
            if ('exercise' in prep) {
              const r = resolveOvercharge({ battle, cardId, exercise: prep.exercise, correct: true, usedHint: false, responseMs: 1000, srsState: states.get(prep.unitId)!, sessionId: session.id, now: NOW })
              states.set(prep.unitId, r.srsState)
              logs.push(r.log)
              recent.push(prep.exercise.id)
              battle = r.battle
            }
          } else {
            battle = playCard(battle, cardId, null, {})
          }
        }
        if (!battle.result) battle = endTurn(battle)
      }
      expect(battle.result).toBe('victory') // 确定性：若失败换 session.seed 重试（结果稳定）
      playerHp = battle.player.hp
      if (battleIndex === 0) relics = ['fluency-charm']
    }

    // 语音不可用 → 全程无听音题（等强度替代为拼写）
    expect(logs.every(l => l.kind !== 'vocab-listening-spelling')).toBe(true)
    // 到期单位产生了计分日志
    expect(logs.some(l => l.graded && l.grade === 'good')).toBe(true)
    // 直接出牌不产生任何日志：日志数 = 教学数 + 过载数
    const overchargeCount = logs.length - logsBeforeBattle
    expect(overchargeCount).toBeGreaterThan(0)
  })
})
```
